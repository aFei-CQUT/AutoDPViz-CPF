% ====================== 页面布局参数 ======================

% A4纸张，四边距25mm对称布局
\geometry{a4paper,
    left   = 25mm,
    right  = 25mm,
    top    = 25mm,
    bottom = 25mm
}

% 页码样式设置（底部居中）
\pagestyle{fancy}
\fancyhf{} % 清空默认页眉页脚
\fancyfoot[C]{\wuhao\thepage} % 五号字体居中页码
\renewcommand{\headrulewidth}{0pt} % 移除页眉横线

% 全局行距设置（固定15磅基准行距）
\setlength{\baselineskip}{15pt}
\setstretch{1.0}             %行距伸 缩因子保持1.0
